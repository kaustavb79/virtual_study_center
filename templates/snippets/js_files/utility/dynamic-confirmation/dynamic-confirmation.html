This is the example of open confirmation overlay

Here is the html of confirmation overlay

<div class="dynamic-confirmation-container">
    <div class="dcc-wrapper">
        <div class="dcc-content">
            <div class="dcc-close-div">
                <button class="dcc-close-btn" tabindex=0>×</button>
            </div>
            <div class="dcc-header-container">
                ${dcc_header_icon_div}
                <div class="dcc-header-text-div">${data.head}</div>
            </div>
            <div class="dcc-body-container">
                <div class="dcc-body-div">
                    ${data.body}
                </div>
            </div>
            <div class="dcc-footer-container">
                <div class="dcc-footer-controls-div">
                    <button class="dcc-submit-btn" tabindex=0>${data.submit_text}</button>
                    <button class="dcc-cancel-btn" tabindex=0>${data.cancel_text}</button>
                </div>
            </div>
        </div>
    </div>
</div>

Pass an object to dynamic_confirmation(); function to open it.

You can pass given keys, values pair to customize confirmation overlay

        // dynamic_confirmation integration start
        // create configuration object for confirmation overlay
        var confirmation_obj = {
            "icon":"high_risk",
            "head":"Are you sure?",
            "body":"Once you delete the job you never get it back!
                    Once you delete the job you never get it back!",
            "width":"300px",
            "action_type":"delete",
            "action_yes":action_yes,
            "action_no":action_no,
            "action_close":action_close,
            "position":"top",
            "cancel_text":"No",
            "submit_text":"Yes",
            "auto_close":"true"
        }

        // call dynamic_confirmation function from dynamic-confirmation.js
        dynamic_confirmation(confirmation_obj);

        function action_yes(){
            //here define the code execute on action_yes
           after_dynamic_confirm_fn(after_dynamic_confirm_data);
        }
        function action_no(){
            //here define the code execute on action_no
            alert('your press no');
        }
        function action_close(){
            //here define the code execute on action_close
            alert('your press close');
        }


        function after_dynamic_confirm_fn(after_dynamic_confirm_data){
            alert(JSON.stringify(after_dynamic_confirm_data));
        }
